// Mathematical Stochastic Engine Implementation

function executeStochasticEngine() {

// Rate Limit Logic Gate via LocalStorage

const todayStr = new Date().toISOString().split('T')[0];

let runsCount = parseInt(localStorage.getItem('sim_runs_count_' + todayStr) || '0');

if (runsCount >= 2) {

alert("Session Guard Notice: You have reached the local daily capacity calculation limits (2 evaluations per browser session). Reset your cache or try again tomorrow.");

return;

}

localStorage.setItem('sim_runs_count_' + todayStr, (runsCount + 1).toString());


// Collect variables

const chanVol = parseFloat(document.getElementById('chan-vol').value);

const chanAht = parseFloat(document.getElementById('chan-aht').value);

const chanWrap = parseFloat(document.getElementById('chan-wrap').value);

const t1Pct = parseFloat(document.getElementById('tier-1-pct').value) / 100;

const t2Pct = parseFloat(document.getElementById('tier-2-pct').value) / 100;

const t3Pct = parseFloat(document.getElementById('tier-3-pct').value) / 100;


const shiftHours = parseInt(document.getElementById('shift-hours').value);

const slaWaitSec = parseFloat(document.getElementById('sla-wait').value) * 60;

const peakMultiplier = parseFloat(document.getElementById('peak-multiplier').value);

const shrinkageMins = parseFloat(document.getElementById('shrinkage-mins').value);

const absenceProb = parseFloat(document.getElementById('absence-prob').value) / 100;

const internMod = parseFloat(document.getElementById('intern-modifier').value);


// Fetch Configuration matrices

const configA = {

sup: parseInt(document.getElementById('config-a-sup').value),

sen: parseInt(document.getElementById('config-a-sen').value),

int: parseInt(document.getElementById('config-a-int').value)

};

const configB = {

sup: parseInt(document.getElementById('config-b-sup').value),

sen: parseInt(document.getElementById('config-b-sen').value),

int: parseInt(document.getElementById('config-b-int').value)

};


const minutesTotal = shiftHours * 60;

const internPenalty = 2 - internMod;


// Simple Multi-run Simulation modeling

function runSimulationModel(cfg) {

let totalHandled = 0;

let totalSlaPass = 0;

let totalAbandoned = 0;

let cumulativeWait = 0;

let totalBusyMinutes = 0;


let tierWaits = { 1: 0, 2: 0, 3: 0 };

let tierCounts = { 1: 0, 2: 0, 3: 0 };

let timelineQueuePoints = new Array(minutesTotal).fill(0);


// Run 12 independent iterations (Stochastic Loops)

for (let run = 0; run < 12; run++) {

let activeSeniors = cfg.sen;

let activeInterns = cfg.int;


// Apply Sick Leave Roulette

for (let i = 0; i < cfg.sen; i++) if (Math.random() < absenceProb) activeSeniors--;

for (let i = 0; i < cfg.int; i++) if (Math.random() < absenceProb) activeInterns--;

let availableThreads = activeSeniors + activeInterns;

let agentPool = [];

for(let i=0; i<activeSeniors; i++) agentPool.push({ type: 'senior', busyUntil: 0 });

for(let i=0; i<activeInterns; i++) agentPool.push({ type: 'intern', busyUntil: 0 });


let backQueue = [];


// Minute by minute chronological processing pipeline

for (let min = 0; min < minutesTotal; min++) {

// Check for shrinkage hours window

let isLunch = (min >= 180 && min <= 300);

let localMultiplier = isLunch ? peakMultiplier : 1.0;

let activeCapacity = isLunch ? Math.ceil(availableThreads * 0.66) : availableThreads;


// Calculate Lambda per minute interval

let baseLambdaPerMin = chanVol / 60;

let minLambda = baseLambdaPerMin * localMultiplier;


// Poisson Knuth Seeding process

let L = Math.exp(-minLambda);

let k = 0;

let p = 1;

do { k++; p *= Math.random(); } while (p > L);

let arrivals = k - 1;


// Route arrivals into system arrays

for (let a = 0; a < arrivals; a++) {

let randTier = Math.random();

let tier = 1;

if (randTier > t1Pct && randTier <= (t1Pct + t2Pct)) tier = 2;

else if (randTier > (t1Pct + t2Pct)) tier = 3;


backQueue.push({ arrivalTime: min, tier: tier });

}


// Process Queue tracking states

let freeAgents = agentPool.filter(ag => ag.busyUntil <= min);

// Dynamic shrinkage reduction modeling

if (freeAgents.length > activeCapacity) { freeAgents = freeAgents.slice(0, activeCapacity); }


while (backQueue.length > 0 && freeAgents.length > 0) {

let currentCase = backQueue.shift();

let assignedAgent = freeAgents.shift();


let waitTime = min - currentCase.arrivalTime;

cumulativeWait += waitTime;

tierWaits[currentCase.tier] += waitTime;

tierCounts[currentCase.tier]++;


if (waitTime * 60 <= slaWaitSec) totalSlaPass++;

totalHandled++;


// Compute handling duration loads

let multiplier = 1.0;

if (currentCase.tier === 2) multiplier = 1.8;

if (currentCase.tier === 3) multiplier = 3.5;


let baseTaskTime = ((chanAht + chanWrap) / 60) * multiplier;

if (assignedAgent.type === 'intern') {

if (currentCase.tier === 3) {

baseTaskTime *= (internPenalty * 2.0); // Massive penalty on technical tracks

} else {

baseTaskTime *= internPenalty;

}

}


assignedAgent.busyUntil = min + Math.ceil(baseTaskTime);

totalBusyMinutes += baseTaskTime;

}


// Process impatience threshold dropouts

let originalQueueLength = backQueue.length;

backQueue = backQueue.filter(caseItem => {

let itemWait = min - caseItem.arrivalTime;

if (itemWait > 15) { // 15 Minute patience horizon barrier

totalAbandoned++;

return false;

}

return true;

});


timelineQueuePoints[min] += backQueue.length;

}

}


// Average profiles out over the 12 loops

const totalCalculatedCases = totalHandled + totalAbandoned || 1;

return {

sla: (totalSlaPass / totalCalculatedCases) * 100,

avgWait: (cumulativeWait / totalCalculatedCases),

dropRate: (totalAbandoned / totalCalculatedCases) * 100,

occupancy: Math.min(((totalBusyMinutes / 12) / (minutesTotal * (cfg.sen + cfg.int || 1))) * 100, 99.2),

t1Wait: tierCounts[1] ? (tierWaits[1] / tierCounts[1]) : 0,

t2Wait: tierCounts[2] ? (tierWaits[2] / tierCounts[2]) : 0,

t3Wait: tierCounts[3] ? (tierWaits[3] / tierCounts[3]) : 0,

chartData: timelineQueuePoints.map(p => p / 12)

};

}


const resultsA = runSimulationModel(configA);

const resultsB = runSimulationModel(configB);

lastSimulationResults = { a: resultsA, b: resultsB, metadata: { t1: t1Pct*100, t2: t2Pct*100, t3: t3Pct*100, sla: document.getElementById('sla-wait').value, configA, configB } };


// Update UI Interface elements

document.getElementById('res-sla-a').innerText = resultsA.sla.toFixed(1) + '%';

document.getElementById('res-sla-b').innerText = resultsB.sla.toFixed(1) + '%';

document.getElementById('res-wait-a').innerText = resultsA.avgWait.toFixed(1) + 'm';

document.getElementById('res-wait-b').innerText = resultsB.avgWait.toFixed(1) + 'm';

document.getElementById('res-drop-a').innerText = resultsA.dropRate.toFixed(1) + '%';

document.getElementById('res-drop-b').innerText = resultsB.dropRate.toFixed(1) + '%';


document.getElementById('bar-occ-a').style.width = resultsA.occupancy.toFixed(0) + '%';

document.getElementById('bar-occ-a').innerText = resultsA.occupancy.toFixed(0) + '%';

document.getElementById('bar-occ-b').style.width = resultsB.occupancy.toFixed(0) + '%';

document.getElementById('bar-occ-b').innerText = resultsB.occupancy.toFixed(0) + '%';


document.getElementById('tier-1-metrics').innerText = resultsA.t1Wait.toFixed(1) + 'm vs ' + resultsB.t1Wait.toFixed(1) + 'm';

document.getElementById('tier-2-metrics').innerText = resultsA.t2Wait.toFixed(1) + 'm vs ' + resultsB.t2Wait.toFixed(1) + 'm';

document.getElementById('tier-3-metrics').innerText = resultsA.t3Wait.toFixed(1) + 'm vs ' + resultsB.t3Wait.toFixed(1) + 'm';


// Generate Automated Layman Translation Strings

let verdict = "";

if (resultsB.sla < resultsA.sla && configB.int > configA.int) {

verdict = "⚠️ INTERN FRAGILITY TRAP ENCOUNTERED: Your experimental profile increases un-trained junior headcount layers while decreasing full-time slots. While you have more active staff threads, your high proportion of Tier 2 and Tier 3 complex tickets creates immediate bottlenecks. Intern processing times and elongated CRM notes logging windows block frontline channels, forcing simple Tier 1 calls to choke and causing a severe queue backup.";

} else if (resultsB.occupancy > 88) {

verdict = "🔥 CRITICAL FRONTLINE REDLINE EXHAUSTION: Configuration B forces your active staff threads into a constant, compounding production ceiling. Staff occupancy exceeds safe barriers, causing post-interaction wrap-up times to get compressed, which guarantees CRM documentation inaccuracies, missed escalations, and high employee attrition loops.";

} else if (resultsB.sla >= 90) {

verdict = "⚖ SUSTAINABLE RESOURCE DEPLOYMENT HARMONY: Configuration B hits target performance envelopes effectively. The roster ratio stabilizes service levels, provides safe bandwidth recovery to handle complex escalations, and clears the holding lines cleanly without driving excessive idle capacity constraints.";

} else {

verdict = "📋 CONGESTED TRAFFIC PATTERNS DETECTED: The structural mix handles early processing loads but experiences cascading delays during lunch break windows. The team clears queues during low volume loops but remains vulnerable to simple customer abandonment curves under sudden load spikes.";

}

document.getElementById('verdict-text').innerText = verdict;


// Render Dynamic Time-Series Chart Data

renderEkgChart(resultsA.chartData, resultsB.chartData);


// Populating hidden PDF template DOM

document.getElementById('pdf-meta-sla').innerText = 'Target SLA: Wait Time < ' + document.getElementById('sla-wait').value + '.0 mins';

document.getElementById('pdf-t1-pct').innerText = (t1Pct*100) + '%';

document.getElementById('pdf-t2-pct').innerText = (t2Pct*100) + '%';

document.getElementById('pdf-t3-pct').innerText = (t3Pct*100) + '%';

document.getElementById('pdf-sup-a').innerText = configA.sup;

document.getElementById('pdf-sen-a').innerText = configA.sen;

document.getElementById('pdf-int-a').innerText = configA.int;

document.getElementById('pdf-sup-b').innerText = configB.sup;

document.getElementById('pdf-sen-b').innerText = configB.sen;

document.getElementById('pdf-int-b').innerText = configB.int;


document.getElementById('pdf-sla-val-a').innerText = resultsA.sla.toFixed(1) + '%';

document.getElementById('pdf-sla-val-b').innerText = resultsB.sla.toFixed(1) + '%';

document.getElementById('pdf-wait-val-a').innerText = resultsA.avgWait.toFixed(1) + 'm';

document.getElementById('pdf-wait-val-b').innerText = resultsB.avgWait.toFixed(1) + 'm';

document.getElementById('pdf-drop-val-a').innerText = resultsA.dropRate.toFixed(1) + '%';

document.getElementById('pdf-drop-val-b').innerText = resultsB.dropRate.toFixed(1) + '%';

document.getElementById('pdf-occ-val-a').innerText = resultsA.occupancy.toFixed(1) + '%';

document.getElementById('pdf-occ-val-b').innerText = resultsB.occupancy.toFixed(1) + '%';


document.getElementById('pdf-t1-wait-a').innerText = resultsA.t1Wait.toFixed(1) + 'm';

document.getElementById('pdf-t2-wait-a').innerText = resultsA.t2Wait.toFixed(1) + 'm';

document.getElementById('pdf-t3-wait-a').innerText = resultsA.t3Wait.toFixed(1) + 'm';

document.getElementById('pdf-t1-wait-b').innerText = resultsB.t1Wait.toFixed(1) + 'm';

document.getElementById('pdf-t2-wait-b').innerText = resultsB.t2Wait.toFixed(1) + 'm';

document.getElementById('pdf-t3-wait-b').innerText = resultsB.t3Wait.toFixed(1) + 'm';

document.getElementById('pdf-verdict').innerText = verdict;


// Reveal hidden scoreboard component

const scoreboardSection = document.getElementById('scoreboard-section');

scoreboardSection.classList.remove('hidden');

scoreboardSection.scrollIntoView({ behavior: 'smooth' });

}


function renderEkgChart(dataA, dataB) {

const ctx = document.getElementById('ekgChart').getContext('2d');

if (globalEkgChart) globalEkgChart.destroy();


// Formulate step markers matching hourly divisions

let labels = [];

for(let i=0; i<dataA.length; i++) {

if(i % 60 === 0) labels.push('Hour ' + (i/60));

else labels.push('');

}


globalEkgChart = new Chart(ctx, {

type: 'line',

data: {

labels: labels,

datasets: [

{ label: 'Configuration A (Baseline)', data: dataA, borderColor: '#64748b', borderWidth: 2, pointRadius: 0, fill: false, tension: 0.1 },

{ label: 'Configuration B (Experimental)', data: dataB, borderColor: '#4f46e5', borderWidth: 2, pointRadius: 0, fill: false, tension: 0.1 }

]

},

options: {

responsive: true,

maintainAspectRatio: false,

scales: {

y: { beginAtZero: true, title: { display: true, text: 'Pending Holding Queue Count', font: { size: 10, weight: 'bold' } } },

x: { grid: { display: false } }

},

plugins: { legend: { labels: { font: { size: 10, weight: 'bold' } } } }

}

});

}
